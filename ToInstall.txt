##BBmap
$ cd software

$ tar xzf BBMap_38.96.tar.gz
create the working folder bbmap. done

## install SWARM
$ cd software
$ git clone https://github.com/torognes/swarm.git
$ cd swarm/src/
$ make
done

## vsearch 
$ cd software
$ wget https://github.com/torognes/vsearch/archive/v2.21.1.tar.gz
$ tar xzf v2.21.1.tar.gz
$ cd vsearch-2.21.1
$ ./autogen.sh
##change PATH by your path (check using pwd)
$ ./configure CFLAGS="-O3" CXXFLAGS="-O3" --prefix=/PATH --libdir=/PATH/software
$ make
$ make install
done

## MAFFT
$ cd software 
$ wget https://mafft.cbrc.jp/alignment/software/mafft-7.490-with-extensions-src.tgz
$ tar xfvz mafft-7.490-with-extensions-src.tgz
$ cd mafft-7.490-with-extensions-src/core
$ make clean
$ make DESTDIR=/PATH/software/mafft-7.490-with-extensions-src/core
$ make install
done

## RAxML
$ cd software
$ git clone https://github.com/stamatak/standard-RAxML.git
$ cd standard-RAxML/
$ make -f Makefile.AVX2.PTHREADS.gcc
done

## EMBOSS 
$ cd software
$ wget ftp://emboss.open-bio.org/pub/EMBOSS/emboss-latest.tar.gz
$ tar -xzvf emboss-latest.tar.gz
$ cd EMBOSS-6.6.0
##change PATH by your path (check using pwd)
$ make DESTDIR=/PATH/software/EMBOSS-6.6.0 -j 4 
$ make install
done
